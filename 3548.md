[Problem: 3548. 等和矩阵分割 II](https://leetcode.cn/problems/equal-sum-grid-partition-ii/description/)

### 方法：枚举+哈希

将分割后的两部分矩阵，分别记作 $P1$ 和 $P2$。

注意，水平/垂直分割都是将一个大矩阵切分为两个小矩阵。因此，垂直分割 **等价于** 将矩阵 **转置** 后进行水平分割，因此只需要写一个处理水平分割的函数。

移除一个单元格后，怎么让该部分继续连通？在一个 $H \times W$ 的矩形中：

- 如果 $H > 1$ 且 $W > 1$：移除 **任何** 一个单元格，剩余部分都一定是连通的
- 如果 $H = 1$（只有一行）：只能移除 **最左端** 或 **最右端** 的单元格，移除中间的会断开
- 如果 $W = 1$（只有一列）：只能移除 **最上端** 或 **最下端** 的单元格
- 如果 $H=1, W=1$：移除唯一的单元格后为空，这种情况在等和条件下不会发生

设整个矩阵的总和为 $S$，分割后的两部分和分别为 $S_1$ 和 $S_2$，也即 $S_1 + S_2 = S$。

- 情况一：不移除单元格，必须满足 $S_1 = S_2$。
- 情况二：从 $P1$ 移除一个值 $x$。使得 $S_1 - x = S_2$。变形得 $x = S_1 - S_2$。此时，需要满足： $S_1 > S_2$，且 $P1$ 中存在值为 $x$ 的单元格，且该位置满足连通性。
- 情况 C：从 $P2$ 移除一个值 $y$。使得 $S_2 - y = S_1$。变形得 $y = S_2 - S_1$。此时，需要满足： $S_2 > S_1$，且 $P2$ 中存在值为 $y$ 的单元格，且该位置满足连通性。

那么，本题的 **关键问题**：如何动态得知某个子矩阵中存在某个数值？

一个核心点：假设我们是水平分割，且从上往下移动分割线。第一次移动后，上方高度为 $1$ 的矩形是 $grid[0][.]$，第二次移动后变为高度 $2$ 的矩形 $grid[0,1][.]$。

这两个矩形只相差一行 $grid[1][.]$ 的所有元素！也即，移动一次后，上方矩阵的新增元素非常容易维护！与此同时，我们还希望快速查找上方矩阵是否存在某个值。

因此，答案呼之欲出，采用一个数据结构：**哈希表** $HashMap$。

具体来说，使用两个 $Map$ 动态维护 $P1$ 和 $P2$ 中所有元素的频率。当水平分割线下移一行时，只需将 **这一行** 的元素从 $P2$ 移动到 $P1$，更新哈希表和累加和。

总的来看，本题有三个技巧：

1. 转置矩阵，将垂直分割的判断转化为水平分割；
2. 使用 $Map$ 实时更新子矩形的元素，避免重复遍历子矩阵；
3. 通过矩阵的结构，将连通性判断简化为对矩形边界的检查。

代码如下，已附加注释：

```Python
# python
class Solution:
    def canPartitionGrid(self, grid: List[List[int]]) -> bool:
        # 所有元素总和
        total_sum = sum(sum(row) for row in grid)

        def check_horizontal(mat):
            m = len(mat)
            n = len(mat[0])
            
            # 记录下方矩形的所有元素频率
            p2_counts = Counter()
            for r in range(m):
                for c in range(n):
                    p2_counts[mat[r][c]] += 1
            
            # 上方矩阵
            p1_counts = Counter()
            p1_sum = 0
            
            # 遍历所有可能的水平分割线
            for i in range(m - 1):
                # 将第 i 行从 P2 转移到 P1
                for j in range(n):
                    val = mat[i][j]
                    p1_sum += val
                    p1_counts[val] += 1
                    p2_counts[val] -= 1
                    if p2_counts[val] == 0:
                        del p2_counts[val]
                
                p2_sum = total_sum - p1_sum
                
                # 直接相等
                if p1_sum == p2_sum:
                    return True
                
                # 尝试从 P1 中移除一个元素
                target1 = p1_sum - p2_sum
                if target1 > 0:
                    h1, w1 = i + 1, n
                    # 连通性检查
                    if h1 > 1 and w1 > 1:
                        if target1 in p1_counts:
                            return True
                    elif h1 == 1 and w1 == 1:
                        if mat[0][0] == target1:
                            return True
                    elif h1 == 1: # 只有一行，检查左右端点
                        if mat[0][0] == target1 or mat[0][w1-1] == target1:
                            return True
                    elif w1 == 1: # 只有一列，检查上下端点
                        if mat[0][0] == target1 or mat[i][0] == target1:
                            return True
                
                # 尝试从 P2 中移除一个元素
                target2 = p2_sum - p1_sum
                if target2 > 0:
                    h2, w2 = m - i - 1, n
                    # 连通性检查
                    if h2 > 1 and w2 > 1:
                        if target2 in p2_counts:
                            return True
                    elif h2 == 1 and w2 == 1:
                        if mat[i+1][0] == target2:
                            return True
                    elif h2 == 1: # 只有一行
                        if mat[i+1][0] == target2 or mat[i+1][w2-1] == target2:
                            return True
                    elif w2 == 1: # 只有一列
                        if mat[i+1][0] == target2 or mat[m-1][0] == target2:
                            return True
                        
            return False

        # 检查原始矩阵
        if check_horizontal(grid):
            return True
        
        # 检查转置矩阵（垂直分割）
        grid_T = [list(row) for row in zip(*grid)]
        return check_horizontal(grid_T)
```

```Java
// java
class Solution {
    public boolean canPartitionGrid(int[][] grid) {
        // 所有元素总和
        long totalSum = 0;
        for (int[] row : grid) {
            for (int val : row) {
                totalSum += val;
            }
        }

        if (checkHorizontal(grid, totalSum)) {
            return true;
        }

        // 检查转置矩阵（垂直分割）
        int m = grid.length;
        int n = grid[0].length;
        int[][] gridT = new int[n][m];
        for (int r = 0; r < m; r++) {
            for (int c = 0; c < n; c++) {
                gridT[c][r] = grid[r][c];
            }
        }
        return checkHorizontal(gridT, totalSum);
    }

    private boolean checkHorizontal(int[][] mat, long totalSum) {
        int m = mat.length;
        int n = mat[0].length;

        // 记录下方矩形的所有元素频率
        Map<Integer, Integer> p2Counts = new HashMap<>();
        for (int r = 0; r < m; r++) {
            for (int c = 0; c < n; c++) {
                p2Counts.put(mat[r][c], p2Counts.getOrDefault(mat[r][c], 0) + 1);
            }
        }

        // 上方矩阵
        Map<Integer, Integer> p1Counts = new HashMap<>();
        long p1Sum = 0;

        // 遍历所有可能的水平分割线
        for (int i = 0; i < m - 1; i++) {
            // 将第 i 行从 P2 转移到 P1
            for (int j = 0; j < n; j++) {
                int val = mat[i][j];
                p1Sum += val;
                p1Counts.put(val, p1Counts.getOrDefault(val, 0) + 1);
                p2Counts.put(val, p2Counts.get(val) - 1);
                if (p2Counts.get(val) == 0) {
                    p2Counts.remove(val);
                }
            }

            long p2Sum = totalSum - p1Sum;

            // 直接相等
            if (p1Sum == p2Sum) {
                return true;
            }

            // 尝试从 P1 中移除一个元素
            long target1 = p1Sum - p2Sum;
            if (target1 > 0 && target1 <= Integer.MAX_VALUE) {
                int t1 = (int) target1;
                int h1 = i + 1;
                int w1 = n;
                // 连通性检查
                if (h1 > 1 && w1 > 1) {
                    if (p1Counts.containsKey(t1))
                    return true;
                } else if (h1 == 1 && w1 == 1) {
                    if (mat[0][0] == t1)
                    return true;
                } else if (h1 == 1) { // 只有一行，检查左右端点
                    if (mat[0][0] == t1 || mat[0][w1 - 1] == t1)
                    return true;
                } else if (w1 == 1) { // 只有一列，检查上下端点
                    if (mat[0][0] == t1 || mat[i][0] == t1)
                    return true;
                }
            }

            // 尝试从 P2 中移除一个元素
            long target2 = p2Sum - p1Sum;
            if (target2 > 0 && target2 <= Integer.MAX_VALUE) {
                int t2 = (int) target2;
                int h2 = m - i - 1;
                int w2 = n;
                // 连通性检查
                if (h2 > 1 && w2 > 1) {
                    if (p2Counts.containsKey(t2))
                    return true;
                } else if (h2 == 1 && w2 == 1) {
                    if (mat[i + 1][0] == t2)
                    return true;
                } else if (h2 == 1) { // 只有一行
                    if (mat[i + 1][0] == t2 || mat[i + 1][w2 - 1] == t2)
                    return true;
                } else if (w2 == 1) { // 只有一列
                    if (mat[i + 1][0] == t2 || mat[m - 1][0] == t2)
                    return true;
                }
            }
        }
        return false;
    }
}
```

- 时间复杂度： $O(m\times n)$，其中 $m,n$ 是矩阵 $grid$ 的行数、列数，计算一次总和，水平扫描时每一行都要移动，转置矩阵
- 空间复杂度： $O(m\times n)$，表示哈希 $p1,p2$ 以及转移矩阵 $gridT$ 的大小
